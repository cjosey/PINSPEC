207 inline int Isotope::getEnergyGridIndex(float energy) const {
208 
209  int index;
210 
211  /* Compute the lethargy */
212  float lethargy = log10(energy);
213 
214  /* If the energy is outside of the grid, pin the energy to the max/min */
215  if (lethargy > _end_lethargy)
216  index = _num_energies - 1;
217  else if (lethargy < _start_lethargy)
218  index = 0;
219  /* If the lethargy is within the grid, comput the index */
220  else
221  index = int(floor((lethargy - _start_lethargy) / _delta_lethargy));
222 
223  return index;
224 }
