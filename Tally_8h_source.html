1621 inline int Tally::getBinIndex(double sample) {
1622 
1623  if (_num_bins == 0)
1624  log_printf(ERROR, "Cannot return a bin index for Tally %s since "
1625  "the bins have not yet been created", _tally_name);
1626 
1627  /* Set index to infinity to begin with */
1628  int index = std::numeric_limits<int>::infinity();
1629 
1630  /* if the sample is equal to the last bin edge, return the last bin */
1631  if (sample == _edges[_num_bins])
1632  return _num_bins-1;
1633 
1634  /* Logarithmically spaced bins */
1635  if (_bin_spacing == LOGARITHMIC)
1636  index = int((log10(sample) - log10(_edges[0])) / _bin_delta);
1637 
1638  /* Equally spaced bins */
1639  else if (_bin_spacing == EQUAL)
1640  index = int((sample - _edges[0]) / _bin_delta);
1641 
1642  /* If the bin_type == OTHER then the bin edges were not generated by
1643  * generateEqualBinEdges, so use a binary search to find the bin */
1644  else
1645  index = findUpperIndex(_edges, _num_bins, 0, sample) - 1;
1646 
1647  /* If this sample was not contained within a bin set index to infinity*/
1648  if (index > _num_bins)
1649  index = std::numeric_limits<int>::infinity();
1650 
1651  return index;
1652 }
