01621 inline int Tally::getBinIndex(double sample) {
01622 
01623     if (_num_bins == 0)
01624         log_printf(ERROR, "Cannot return a bin index for Tally %s since "
01625            "the bins have not yet been created", _tally_name);
01626 
01627     /* Set index to infinity to begin with */
01628     int index = std::numeric_limits<int>::infinity();
01629 
01630     /* if the sample is equal to the last bin edge, return the last bin */
01631     if (sample == _edges[_num_bins])
01632         return _num_bins-1;
01633 
01634     /* Logarithmically spaced bins */
01635     if (_bin_spacing == LOGARITHMIC)
01636         index = int((log10(sample) - log10(_edges[0])) / _bin_delta);
01637 
01638     /* Equally spaced bins */
01639     else if (_bin_spacing == EQUAL)
01640         index = int((sample - _edges[0]) / _bin_delta);
01641 
01642     /* If the bin_type == OTHER then the bin edges were not generated by
01643      * generateEqualBinEdges, so use a binary search to find the bin */
01644     else
01645         index = findUpperIndex(_edges, _num_bins, 0, sample) - 1;
01646 
01647     /* If this sample was not contained within a bin set index to infinity*/
01648     if (index > _num_bins)
01649         index = std::numeric_limits<int>::infinity();
01650 
01651     return index;
01652 }
