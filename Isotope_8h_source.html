235 inline int Isotope::getEnergyGridIndex(float energy) const {
236 
237  int index;
238 
239  /* Compute the lethargy */
240  float lethargy = log10(energy);
241 
242  /* If the energy is outside of the grid, pin the energy to the max/min */
243  if (lethargy > _end_lethargy)
244  index = _num_energies - 1;
245  else if (lethargy < _start_lethargy)
246  index = 0;
247  /* If the lethargy is within the grid, comput the index */
248  else
249  index = int(floor((lethargy - _start_lethargy) / _delta_lethargy));
250 
251  return index;
252 }
